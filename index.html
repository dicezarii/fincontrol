// ==================== SETTINGS ==================== let currentEditingBank = null; let currentCatType = 'expense'; function renderSettings() { renderBanksList(); renderCategoriesList(); } function renderBanksList() { const container = document.getElementById('banks-list-settings'); if (banks.length === 0) { container.innerHTML = `
Nenhum banco cadastrado.
Adicione seu primeiro banco acima.
`; return; } container.innerHTML = banks.map((bank, index) => { const bankTx = expenses.filter(e => e.bankId == bank.id); const income = bankTx.filter(e => e.type === 'income').reduce((s, e) => s + e.value, 0); const expense = bankTx.filter(e => e.type === 'expense').reduce((s, e) => s + e.value, 0); const currentBalance = (bank.initial || 0) + income - expense; return `
🏦
${bank.name}
${bankTx.length} transações
R$ ${currentBalance.toFixed(2)}
Inicial: R$ ${(bank.initial || 0).toFixed(2)}
`; }).join(''); } function renderCategoriesList() { // Despesas const expContainer = document.getElementById('expense-cats-list'); expContainer.innerHTML = Object.entries(categories.expense || {}).map(([id, name]) => ` ${name} × `).join(''); // Receitas const incContainer = document.getElementById('income-cats-list'); incContainer.innerHTML = Object.entries(categories.income || {}).map(([id, name]) => ` ${name} × `).join(''); } // ==================== BANKS MANAGEMENT ==================== function openBankSheet() { currentEditingBank = null; document.getElementById('bank-sheet-title').textContent = 'Novo Banco'; document.getElementById('bank-name').value = ''; document.getElementById('bank-initial').value = ''; document.getElementById('bank-color').value = '#6366f1'; document.getElementById('btn-delete-bank').style.display = 'none'; openSheet('bank-sheet'); } function editBank(index) { currentEditingBank = index; const bank = banks[index]; document.getElementById('bank-sheet-title').textContent = 'Editar Banco'; document.getElementById('bank-name').value = bank.name; document.getElementById('bank-initial').value = bank.initial || 0; document.getElementById('bank-color').value = bank.color || '#6366f1'; document.getElementById('btn-delete-bank').style.display = 'block'; openSheet('bank-sheet'); } function saveBank() { const name = document.getElementById('bank-name').value.trim(); const initial = parseFloat(document.getElementById('bank-initial').value) || 0; const color = document.getElementById('bank-color').value; if (!name) { showToast('Digite o nome do banco'); return; } if (currentEditingBank !== null) { banks[currentEditingBank] = { ...banks[currentEditingBank], name, initial, color }; showToast('Banco atualizado'); } else { banks.push({ id: Date.now(), name, initial, color }); showToast('Banco adicionado'); } localStorage.setItem('fc_banks', JSON.stringify(banks)); closeAllSheets(); renderSettings(); updateUI(); } function deleteCurrentBank() { if (currentEditingBank === null) return; if (!confirm('Tem certeza? Todas as transações deste banco ficarão sem vínculo.')) return; banks.splice(currentEditingBank, 1); localStorage.setItem('fc_banks', JSON.stringify(banks)); showToast('Banco removido'); closeAllSheets(); renderSettings(); updateUI(); } // ==================== CATEGORIES MANAGEMENT ==================== function openCategorySheet(type) { currentCatType = type; document.getElementById('cat-emoji-input').value = ''; document.getElementById('cat-name-input').value = ''; openSheet('category-sheet'); } function saveCategory() { const emoji = document.getElementById('cat-emoji-input').value.trim(); const name = document.getElementById('cat-name-input').value.trim(); if (!emoji || !name) { showToast('Preencha ícone e nome'); return; } const id = name.toLowerCase().replace(/\s+/g, '_'); if (!categories[currentCatType]) categories[currentCatType] = {}; categories[currentCatType][id] = `${emoji} ${name}`; localStorage.setItem('fc_cats', JSON.stringify(categories)); showToast('Categoria adicionada'); closeAllSheets(); renderSettings(); } function deleteCategory(type, id) { if (!confirm('Remover esta categoria?')) return; delete categories[type][id]; localStorage.setItem('fc_cats', JSON.stringify(categories)); renderSettings(); } // ==================== DATA EXPORT/IMPORT ==================== function exportData() { const data = { expenses, banks, loans, categories, exportDate: new Date().toISOString(), version: '1.0' }; const blob = new Blob([JSON.stringify(data, null, 2)], { type: 'application/json' }); const url = URL.createObjectURL(blob); const a = document.createElement('a'); a.href = url; a.download = `fincontrol_backup_${new Date().toISOString().split('T')[0]}.json`; document.body.appendChild(a); a.click(); document.body.removeChild(a); URL.revokeObjectURL(url); showToast('Backup exportado!'); } function importData(input) { const file = input.files[0]; if (!file) return; const reader = new FileReader(); reader.onload = function(e) { try { const data = JSON.parse(e.target.result); if (!confirm(`Isso substituirá todos os dados atuais.\n\nDespesas: ${data.expenses?.length || 0}\nBancos: ${data.banks?.length || 0}\nEmpréstimos: ${data.loans?.length || 0}\n\nContinuar?`)) { return; } expenses = data.expenses || []; banks = data.banks || []; loans = data.loans || []; categories = data.categories || defaultCats; localStorage.setItem('fc_expenses', JSON.stringify(expenses)); localStorage.setItem('fc_banks', JSON.stringify(banks)); localStorage.setItem('fc_loans', JSON.stringify(loans)); localStorage.setItem('fc_cats', JSON.stringify(categories)); updateUI(); renderSettings(); showToast('Dados importados com sucesso!'); } catch (err) { showToast('Erro ao importar arquivo'); console.error(err); } input.value = ''; }; reader.readAsText(file); } function clearAllData() { if (!confirm('⚠️ ATENÇÃO!\n\nIsso apagará TODOS os dados permanentemente.\n\nDigite "APAGAR" para confirmar.')) return; const confirmation = prompt('Digite "APAGAR" para confirmar:'); if (confirmation !== 'APAGAR') { showToast('Operação cancelada'); return; } expenses = []; banks = []; loans = []; categories = JSON.parse(JSON.stringify(defaultCats)); localStorage.removeItem('fc_expenses'); localStorage.removeItem('fc_banks'); localStorage.removeItem('fc_loans'); localStorage.removeItem('fc_cats'); updateUI(); renderSettings(); showView('main-view'); showToast('Todos os dados foram apagados'); }
